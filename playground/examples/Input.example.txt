class InputExample extends React.Component {

    constructor(){
        super();
        this.onChange = this.onChange.bind(this);
        this.validate = this.validate.bind(this);
        this.onClipboardButtonClick = this.onClipboardButtonClick.bind(this);

        this.state = {
            status: '',
            message: '',
            value: 'copy value!!',
            icon: 'search'
        }
    }

    render(){

        const pers = [
            {
                value: 'haha',
                label: 'hihi'
            },
            {
                value: 'test',
                label: 'testLabel'
            }    
        ];

        return(
            <div>
                <Input value={this.state.value} onChange={this.onChange} personalizationTags={pers} emoji icon="check" description="hahah" status="success" type="text" placeholder="max length 5" label="Some label:"/>
                <br/>
                <Input
                    onChange={this.onChange}
                    value="password"
                    type="password"
                    label="Password:"
                    labelTop
                />
                <br/>
                <Input value={this.state.value} onClipboardButtonClick={this.onClipboardButtonClick} clipboardButton="Copy Value" validate={this.validate} icon={this.state.icon} status={this.state.status} description={this.state.message} onChange={this.onChange} placeholder="placeholder" className=".input-field" autocomplete/>
                <br/>
                <Input onChange={this.onChange} disabled value="disabled" description="This input is disabled"/>
            </div>
        );
    }

    onChange(e, data) {
        console.log(e.target);
        this.setState({value: data.value})
    }

    validate(data){
        console.log(data.value);
        if (data.value){
            if (data.value.length < 5){
                this.setState({status: 'error', message: 'Input should have 5 or more characters', icon: 'clear'});
            } else {
                this.setState({status: 'success', message: '', icon: 'check'})
            }
        } else {
            this.setState({status: 'error', message: 'This input field is required', icon: 'clear'});
        }
    }

    onClipboardButtonClick(e, data) {
        alert('Copied: ' + data.value);
    }

}
