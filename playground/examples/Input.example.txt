class InputExample extends React.Component {

    constructor(){
        super();
        this.onChange = this.onChange.bind(this);

        this.state = {
            status: '',
            message: ''
        }
    }

    render(){
        return(
            <div>
                <Input onChange={this.onChange} type="text" maxlength={5} placeholder="max length 5" label="Some label:"/>
                <br/>
                <Input
                    onChange={this.onChange}
                    value="password"
                    type="password"
                    label="Password:"
                    description="This has autofocus on page load"
                />
                <br/>
                <Input status={this.state.status} description={this.state.message} onChange={this.onChange} placeholder="placeholder" className=".input-field" autocomplete/>
                <br/>
                <Input onChange={this.onChange} disabled value="disabled" description="This input is disabled"/>
            </div>
        );
    }

    onChange(e, data) {
        console.log(e.target);
        if (data.value.length < 5){
            this.setState({status: 'error', message: 'Input should have 5 or more characters'});
        } else {
            this.setState({status: '', message: ''})
        }
    }

}
