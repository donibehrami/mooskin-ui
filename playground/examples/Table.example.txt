    logIt(a, b, order, sortBy) {

        let comparison = 0;

        if (order === 'desc'){

            console.log('desc called');

            if (a[sortBy] > b[sortBy]){
                comparison = -1;
            } else if (a[sortBy] < b[sortBy]){
                comparison = 1;
            }
            return comparison;
        } else if (order === 'asc') {

            console.log('asc called');

            if (a[sortBy] < b[sortBy]){
                comparison = -1;
            } else if (a[sortBy] > b[sortBy]){
                comparison = 1;
            }
            return comparison;
        }

    }
